const  int n = 3;


int main()
{
	double  A[n][n + 1] = { {2,5,4,1},{7,3,1,3},{7,4,2,8} };
	int i = 0, k = 0, j = n; // i-Перебирання стрічок, k-переберання діагональних елеменів, j-робота
	for (; k < n - 1; k++)
	{
		for (j=n; j >= k; A[k][j--] /= A[k][k]);// ділення на А[k][k]
		for (i = k + 1; i < n; i++)
			for (j = n; j >= k; j--)
				A[i][j] -= A[k][j] * A[i][k];
	}
	for (i = 0; i < n + 1; i++)
	{   
		cout << "\n";
		for (j = 0; j < n+1; j++)
			cout <<"\t"<<A[i][j];
	}
    return 0;
}
